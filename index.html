import { humanReadableNumber, pluralize, _ } from 'camote-utils';

// Example 1: Converting big numbers to readable format
// Input: [1234567, 2345678, 3456789]
// Steps: 1. Create a chain of operations
//        2. Convert each number to readable format (1.23M)
//        3. Join them with commas
const bigNumbers = [1234567, 2345678, 3456789];
const readableNumbers = _.chain(bigNumbers)
  .map(humanReadableNumber)     // Converts to: ["1.23M", "2.35M", "3.46M"]
  .join(', ')                   // Joins with commas
  .value();                     // Ends chain and returns result
console.log(readableNumbers);   // Output: "1.23M, 2.35M, 3.46M"

// Example 2: Simple pluralization
// Shows how words change based on count
const animals = _.map([1, 2, 5], count => {
  const word = pluralize("cat", count);
  return `${count} ${word}`;    // Adds count before each word
});
console.log(animals);           // Output: ["1 cat", "2 cats", "5 cats"]

// Example 3: Custom plural forms
// Some words have special plural forms
const specialCases = [
  { word: "child", count: 2, plural: "children" },
  { word: "person", count: 3, plural: "people" }
];

const result = _.chain(specialCases)
  .map(item => {
    const plural = pluralize(item.word, item.count, item.plural);
    return `${item.count} ${plural}`;  // Adds count for clarity
  })
  .join(" and ")               // Joins with 'and' for readability
  .value();
console.log(result);           // Output: "2 children and 3 people"
